LinearStack.h

template <typename T>  
class LinearStack//顺序栈  
{  
private:  
    int size;//栈的控件  
    int length;//当前存放数据长度  
    T *data;  
public:  
    LinearStack(int size)  
    {  
        this->size = size;  
        this->length = 0;  
        data = new T[size];  
    }  

    ~LinearStack()  
    {  
        delete[] data;  
    }  

    void Creat(T newData[], int length)//批量创建  
    {  
        if (length > size || length <= 0)  
        {  
            throw std::exception("Length error!!!!");  
            return;  
        }  
        this->length = length;  
        for (int i = 0; i < length; i++)  
        {  
            data[i] = newData[i];  
        }  
    }  

    void Push(T newData)//入栈  
    {  
        if (isFull())  
        {  
            throw std::exception("Over Up!!!");  
            return;  
        }  
        data[length++] = newData;  
    }  

    T Pop()//出栈  
    {  
        if (isEmpty())  
        {  
            throw std::exception("Over Bottom!!!");  
            return NULL;  
        }  
        return data[--length];  
    }  

    T GetTop()//返回栈顶元素  
    {  
        return data[length - 1];  
    }  

    bool isEmpty()//判断是否栈空  
    {  
        if (length == 0)  
        {  
            return true;  
        }  
        return false;  
    }  

    bool isFull()//判断是否栈满  
    {  
        if (length == size)  
        {  
            return true;  
        }  
        return false;  
    }  

    void PrintList()//输出  
    {  
        cout << "size=" << size << " length=" << length << endl;  
        for (int i = 0; i < length; i++)  
        {  
            cout << data[i] << " ";  
        }  
        cout << endl;  
        cout << endl;  
    }  
};

NodeStack.h

template <typename T>  
struct Node_Stack//结点  
{  
    T data;  
    Node_Stack *next;  
};  
template <typename T>  
class NodeStack  
{  
private:  
    int length;//链栈长度  
    Node_Stack<T> *head;  
public:  
    NodeStack()  
    {  
        head = new Node_Stack<T>;  
        length = 0;  
        head->next = NULL;  
        head->data = NULL;//初始化头结点  
    }  
    ~NodeStack()  
    {  
        Node_Stack<T> *m;  
        Node_Stack<T> *n;  
        m = head->next;  
        while (m != NULL)  
        {  
            n = m->next;  
            delete m;  
            m = n;  
        }  
        head->next = NULL;  
        head->data = NULL;  
    }  
    void Creat(T *newData, int length)//批量创建  
    {  
        if (length < 0)  
        {  
            throw std::exception("Length Error!!!");  
            return;  
        }  
        this->length = length;  
        Node_Stack<T> *m;  
        m = head;  
        for (int i = 0; i < length; i++)  
        {  
            Node_Stack<T> *n = new Node_Stack<T>;  
            n->data = newData[i];  
            m->next = n;  
            m = m->next;  
        }  
        m->next = NULL;  
        m = NULL;  
    }  
    void Push(T newData)//入栈，直接在头结点处操作  
    {  
        Node_Stack<T> *n=new Node_Stack<T>;  
        n->next = head->next;  
        n->data = newData;  
        head->next = n;  
        length++;  
    }  
    T Pop()//出栈  
    {  
        if (isEmpty())  
        {  
            throw std::exception("List is empty!!!");  
            return NULL;  
        }  
        Node_Stack<T> *temp = head->next;  
        head->next = temp->next;  
        length--;  
        T t = temp->data;//将栈顶元素保存到temp中，然后删除栈顶元素  
        delete temp;  
        return t;  
    }  

    bool isEmpty()//判断是否为空  
    {  
        if (length == 0)  
        {  
            return true;  
        }  
        return false;  
    }  
    T GetTop()//获取栈顶元素  
    {  
        if (isEmpty())  
        {  
            throw std::exception("List i Empty!!!");  
            return NULL;  
        }  
        return head->next->data;  
    }  
    void PrintList()//输出  
    {  
        Node_Stack<T> *m;  
        m = head->next;  
        cout << "size=" << length << endl;  
        for (; m != NULL; m = m->next)  
        {  
            cout << m->data << " ";  
        }  
        cout << endl;  
        cout << endl;  
    }  
};

正如上一篇文章所讲的，栈就是在线性表的基础上添加了一个只能在栈顶操作元素的约束条件。整个代码大多数都是和线性表完全相同，所以写出了线性表，写栈就很简单。
